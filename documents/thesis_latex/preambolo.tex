\newcommand{\chapterstylebox}[1]{%

  % compensazione reale dello spazio del display chapter
  \vspace*{-\baselineskip}
  \vspace*{-\topskip}
  \vspace*{-5pt}

  % --- CAPITOLO (solo nel mainmatter) ---
  \if@mainmatter
    {\centering\scshape\huge CAPITOLO\par}
    \vspace{10pt}
  \fi

  \centering
  \begin{tikzpicture}
    \def\boxwidth{1\textwidth}
    \def\boxheight{4.5cm}

    % lati e base
    \draw (0,0) -- (0,-\boxheight) -- (\boxwidth,-\boxheight) -- (\boxwidth,0);

    % linea superiore
    \if@mainmatter
      % spezzata + numero
      \draw (0,0) -- (0.45*\boxwidth,0);
      \draw (0.55*\boxwidth,0) -- (\boxwidth,0);
      \node at (0.5*\boxwidth,0) {\Huge\bfseries \thechapter};
    \else
      % chiusa (front/back matter)
      \draw (0,0) -- (\boxwidth,0);
    \fi

    % titolo (sempre)
    \node[text width=0.8\textwidth, align=center]
      at (0.5*\boxwidth,-0.55*\boxheight)
      {\Huge \MakeUppercase{#1}};
  \end{tikzpicture}
}
